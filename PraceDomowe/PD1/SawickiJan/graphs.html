Graphs generated with different libraries.

ggplot_pointchart

library(ggplot2)
library(lattice)

data = data.frame(x = runif(100), y = runif(100))

ggplot(data = data, aes(x = x, y = y)) +
    geom_point()

ggplot_linechart

  ggplot(data = data, aes(x = x, y = y)) +
    geom_line()

ggplot_barchart

  ggplot(data = data, aes(
    x = x, y = y, col = x
  )) +
    geom_bar(stat = "identity")

ggplot_boxchart

  ggplot(data = data, aes(
    x = x, y = y, col = x
  )) + geom_boxplot()
## Warning: Continuous x aesthetic -- did you forget aes(group=...)?

ggplot_densitychart

  ggplot(data = data, aes(x = x)) +
    geom_histogram(binwidth = 1)

ggplot_nochart

  ggplot()

lattice_pointchart

  xyplot(y ~ x, data)

lattice_linechart

  xyplot(y ~ x, data, type = c("smooth"))

lattice_barchart

  barchart(y ~ x, data)

lattice_boxchart

  bwplot(data$x, data)
## Warning in bwplot.numeric(data$x, data): explicit 'data' specification
## ignored

lattice_densitychart

  histogram(data$x)

lattice_nochart

  xyplot(0 ~ 0, data = data)

graphics_pointchart

  plot(data$x, data$y, type = "p")

graphics_linechart

  plot(data$x, data$y, type = "l")

graphics_barchart

  plot(data$x, data$y, type = "h")

graphics_boxchart

  boxplot(data$x, data$y)

graphics_densitychart

  hist(data$x)

graphics_nochart

  plot(data$x, data$y, type = "n")
